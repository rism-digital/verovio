131 class AttDurExtreme : public AttComparison {
132 
133 public:
134  AttDurExtreme(DurExtreme extremeType) : AttComparison(OBJECT)
135  {
136  m_extremeType = extremeType;
137  if (m_extremeType == LONGEST)
138  m_extremeDur = -VRV_UNSET;
139  else
140  m_extremeDur = VRV_UNSET;
141  }
142 
143  virtual bool operator()(Object *object)
144  {
145  if (!object->HasInterface(INTERFACE_DURATION)) return false;
146  DurationInterface *interface = dynamic_cast<DurationInterface *>(object);
147  assert(interface);
148  if (interface->HasDur()) {
149  if ((m_extremeType == LONGEST) && (interface->GetActualDur() < m_extremeDur)) {
150  m_extremeDur = interface->GetActualDur();
151  return true;
152  }
153  else if ((m_extremeType == SHORTEST) && (interface->GetActualDur() > m_extremeDur)) {
154  m_extremeDur = interface->GetActualDur();
155  return true;
156  }
157  }
158  return false;
159  }
160 
161 private:
162  int m_extremeDur;
163  DurExtreme m_extremeType;
164 };
